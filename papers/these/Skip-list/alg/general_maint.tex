   {\size 

		\Statex

			\While{$\lit{true}$}
				\State $\mathit{<}\ms{node,next}\mathit{>} \gets \lit{next\_in\_trav}(\lit{true})$
				\While{$\ms{next \neq \bot}$}
					\If{$\ms{next.del}$}
						\State $\lit{remove}(\ms{node}, \ms{next})$
					\EndIf
					\State $\lit{restructure\_node}(\ms{next})$
					\State $\mathit{<}\ms{node,next}\mathit{>} \gets \lit{next\_in\_trav}(\lit{false})$
				\EndWhile
			\State $\lit{restructure\_structure}(\ms{structure})$
			\EndWhile
   		}\EndPart

		\Part{}\EndPart

	}
