	\line{BA01} \> $\ms{node}$ a record with fields: \\

\bf{operation} $\act{find}(k)_p$ \\
		\line{BA01} \> $\ms{next} \gets \ms{root}$ \\
		\line{BA01} \> \bf{while} $(\lit{true})$ \bf{do} \\
		\line{BA01} \>\> $\ms{curr} \gets \ms{next}$; $\ms{val} \gets \ms{curr.k}$ \\
		\line{BA01} \>\> \bf{if} $(\ms{val} = \ms{k})$ \bf{then} $\lit{break}()$ \bf{end if} \label{sline:find-value} \\
		\line{BA01} \>\> \bf{if} $(\ms{val} > k)$ \bf{then} $\ms{next} \gets \lit{read}(\ms{curr.r})$  \label{sline:find-right} \\
		\line{BA01} \>\>\> \bf{else} $\ms{next} \gets \lit{read}(\ms{curr.\ell})$ \bf{end if} \label{sline:find-left} \\
		\line{BA01} \>\> \bf{if} $(\ms{next} = \bot)$ \bf{then} $\lit{break}()$ \bf{end if} \label{sline:find-bot} \\
		\line{BA01} \> \bf{end while} \\
		\line{BA01} \> $\sf{return}(\ms{curr})$. \label{sline:find-return} \\
		

\bf{operation} $\act{contains}(k)_p$ \\
		\line{BA01} \> $\lit{begin\_transaction}()$ \\
		\line{BA01} \> $\ms{result} \gets \lit{true}$; $\ms{curr} \gets \lit{find}(k)$ \label{sline:cont-find} \\
		\line{BA01} \> \bf{if} $(\ms{curr.k} \neq k)$ \bf{then} $\ms{result} \gets \lit{false}$ \label{sline:cont-neq} \\
		\line{BA01} \>\> \bf{else if} $(\lit{read}(\ms{curr.del}))$ \bf{then} $\ms{result} \gets \lit{false}$ \bf{end if} \label{sline:del-cont} \\
		\line{BA01} \> \bf{end if} \\
		\line{BA01} \> $\lit{try\_to\_commit}()$ \\
		\line{BA01} \> $\sf{return}(\ms{result})$. \label{sline:cont-return} \\

   		
\bf{operation} $\act{insert}(k,v)_p$ \\
		\line{BA01} \> $\lit{begin\_transaction}()$ \\
		\line{BA01} \> $\ms{result} \gets \lit{true}$; $\ms{curr} \gets \lit{find}(k)$  \label{sline:find-ins} \\
		\line{BA01} \> \bf{if} $(\ms{curr.k} = \ms{k})$ \bf{then} \\
		\line{BA01} \>\>\> \bf{if} $(\lit{read}(\ms{curr.del}))$ \bf{then} $\lit{write}(\ms{curr.del}, \lit{false})$ \label{sline:del-false-ins} \label{sline:del-ins} \\
						
		\line{BA01} \>\>\>\> \bf{else} $\ms{result} \gets \lit{false}$ \bf{end if} \label{sline:false-ins} \\
		\line{BA01} \>\> \bf{else} \sf{allocate a new node} \label{sline:new-node} \\
		\line{BA01} \>\>\> $\ms{new.k} \gets \ms{k}$; $\ms{new.v} \gets \ms{v}$ \\
		\line{BA01} \>\>\> \bf{if} $(\ms{curr.k} > \ms{k})$ \bf{then} $\lit{write}(\ms{curr.r},\ms{new})$ \label{sline:start-ins} \\
		\line{BA01} \>\>\>\> \bf{else} $\lit{write}(\ms{curr.\ell}, \ms{new})$ \bf{end if} \label{sline:end-ins} \\
		\line{BA01} \> \bf{end if} \\
		\line{BA01} \> $\lit{try\_to\_commit}()$ \\
		\line{BA01} \> $\sf{return}(\ms{result})$. \label{sline:insert-return} \\
